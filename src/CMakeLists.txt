cmake_minimum_required(VERSION 3.15)

# Enable testing for this directory so CMake generates CTestTestfile.cmake
# This allows CTest to discover all test subdirectories automatically
if(WITH_TESTS OR ENABLE_TESTS)
    enable_testing()
endif()

add_subdirectory(secp256k1)
add_subdirectory(infrastructure)
add_subdirectory(domain)
add_subdirectory(consensus)

# Network module is not compatible with Emscripten
if(NOT CMAKE_SYSTEM_NAME STREQUAL "Emscripten")
    add_subdirectory(network)
endif()

add_subdirectory(database)
add_subdirectory(blockchain)
add_subdirectory(node)

# Node executable is not compatible with Emscripten
if (BUILD_NODE_EXE AND NOT CMAKE_SYSTEM_NAME STREQUAL "Emscripten")
    add_subdirectory(node-exe)
endif()

if (BUILD_C_API)
    add_subdirectory(c-api)
endif()
